<script>
const players = [
{
rank: 1,
name: "Player Name",
position: "QB",
school: "School",
tier: "blue",
height: "6'3",
weight: "220",
strengths: "Elite arm talent, off-platform throws",
weaknesses: "Processing under pressure",
comp: "NFL Comp"
},
{
rank: 2,
name: "Player Name",
position: "EDGE",
school: "School",
tier: "round1",
height: "6'5",
weight: "255",
strengths: "Explosive first step",
weaknesses: "Run discipline",
comp: "NFL Comp"
}
];

function renderBoard() {
const board = document.getElementById("board");
board.innerHTML = "";

const positionFilter = document.getElementById("positionFilter").value;
const tierFilter = document.getElementById("tierFilter").value;

players
.filter(p => (positionFilter === "all" || p.position === positionFilter))
.filter(p => (tierFilter === "all" || p.tier === tierFilter))
.forEach(player => {
const card = document.createElement("div");
card.className = "card";

card.innerHTML = `
<div class="rank">Rank #${player.rank}</div>
<div class="name">${player.name}</div>
<div class="tier ${player.tier}">${player.tier.toUpperCase()}</div>
<div class="meta">${player.position} | ${player.school} | ${player.height} | ${player.weight}</div>

<div class="section-title">Strengths</div>
<div>${player.strengths}</div>

<div class="section-title">Weaknesses</div>
<div>${player.weaknesses}</div>

<div class="section-title">NFL Comp</div>
<div>${player.comp}</div>
`;

board.appendChild(card);
});
}

function sortByRank() {
players.sort((a, b) => a.rank - b.rank);
renderBoard();
}

document.getElementById("positionFilter").addEventListener("change", renderBoard);
document.getElementById("tierFilter").addEventListener("change", renderBoard);

renderBoard();
</script>
